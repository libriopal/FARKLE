/**
 * Numeric representation of die pool entries.
 * 0 = rock blocker placeholder (unplayable).
 * 1–6 = die face values.
 */
export type DiePoolEntry = 0 | 1 | 2 | 3 | 4 | 5 | 6;

/**
 * Numeric representation of wild pool entries.
 * 0 = empty slot (plain die face, no overlay).
 * 1 = Ice Blocker overlay.
 * 2 = Lock Blocker overlay.
 * 3 = Wild Blocker overlay.
 */
export type WildPoolEntry = 0 | 1 | 2 | 3;

/**
 * The six-pool state snapshot for persistence and provably fair verification.
 * Each pool is an ordered array — draw order is index 0 first.
 */
export interface SixPoolSnapshot {
  liveDie: DiePoolEntry[];
  deadDie: DiePoolEntry[];
  liveBlocker: number[];
  deadBlocker: number[];
  liveWild: WildPoolEntry[];
  deadWild: WildPoolEntry[];
}

/** Configuration for a single pool's initial weighting */
export interface PoolWeightConfig {
  /** For wild pool: probability of each WildPoolEntry value */
  wildWeights: { empty: number; ice: number; lock: number; wild: number };
  /** For die pool: count of rock placeholders (0s) mixed in */
  rockCount: number;
}
