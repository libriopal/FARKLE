export class SixPoolManager {
  // Six arrays — Live and Dead for each pool type
  private liveDie: DiePoolEntry[];
  private deadDie: DiePoolEntry[];
  private liveBlocker: number[];   // Rock blocker counts (all 0s in sequence)
  private deadBlocker: number[];
  private liveWild: WildPoolEntry[];
  private deadWild: WildPoolEntry[];

  private csprng: { next: () => number }; // HMAC-SHA256 CSPRNG instance

  constructor(playerCount: number, settings: LobbySettings, csprng: { next: () => number })

  POOL SIZE FORMULAS (derive from grid — Q11 Answer B):
    gridDim = playerCount === 1 ? 7 : playerCount === 2 ? 8 : playerCount === 3 ? 9 : 10
    liveDieSize = Math.ceil((gridDim * gridDim) / 6) * 6
      — Equal distribution: liveDieSize/6 of each face (1–6)
      — Rock placeholders (0s): blockerDensityCount(settings) mixed in
      — Total: liveDieSize entries

    liveBlockerSize = blockerDensityCount(settings) * 2
      — All entries are rock blocker tokens (numeric value: 0)

    liveWildSize = Math.ceil((gridDim * gridDim) / 4)
      — Weighted distribution:
          empty (0):  60% of liveWildSize
          ice (1):    15% of liveWildSize
          lock (2):   15% of liveWildSize
          wild (3):   10% of liveWildSize
      — Round counts to nearest integer, ensure total === liveWildSize

  SHUFFLE: Use Fisher-Yates with csprng.next() for all pools.

  METHODS:

  drawDie(): DiePoolEntry
    — Draws from liveDie[0], shifts array
    — If liveDie.length === 0: replenish('die') first
    — Returns 0 (rock) or 1–6 (die face)

  drawWild(): WildPoolEntry
    — Draws from liveWild[0], shifts array
    — If liveWild.length === 0: replenish('wild') first
    — Returns 0 (empty), 1 (ice), 2 (lock), 3 (wild)

  drawBlocker(): number
    — Draws from liveBlocker[0], shifts array
    — If liveBlocker.length === 0: replenish('blocker') first

  replenish(pool: 'die' | 'wild' | 'blocker'): void
    — Promotes dead pool to live pool
    — Creates new dead pool of same size
    — Shuffles new live pool with csprng
    — If dead pool also empty: create fresh pool from scratch

  remainingDie(): number  — returns liveDie.length
  remainingWild(): number — returns liveWild.length
  snapshot(): SixPoolSnapshot — returns copy of all 6 arrays for verification

  PRIVATE:
  private shuffle<T>(arr: T[]): T[]  — Fisher-Yates, uses csprng.next()
  private buildDiePool(size: number, rockCount: number): DiePoolEntry[]
  private buildWildPool(size: number): WildPoolEntry[]
}

/**
 * Scans the grid and returns the count of cells with state 'WILD'.
 * Called after every cascade completion.
 * When result >= WILD_BLOCKER_FRENZY_COUNT, Wild Scatter triggers.
 * @returns Number of Wild Blocker cells currently on the grid
 */
export function scanForWilds(grid: Cell[][]): number {
  let count = 0;
  for (let r = 0; r < grid.length; r++)
    for (let c = 0; c < grid[0].length; c++)
      if (grid[r][c].state === 'WILD') count++;
  return count;
}

/**
 * Returns the positions and face values of all Wild Blocker cells.
 * Used by WILD_SCATTER action to determine which faces to destroy.
 */
export function getWildCells(grid: Cell[][]): Array<{ row: number; col: number; face: DieFace }> {
  const wilds: Array<{ row: number; col: number; face: DieFace }> = [];
  for (let r = 0; r < grid.length; r++)
    for (let c = 0; c < grid[0].length; c++) {
      const cell = grid[r][c];
      if (cell.state === 'WILD' && cell.face !== null)
        wilds.push({ row: r, col: c, face: cell.face });
    }
  return wilds;
}

/**
 * Executes Wild Scatter detonation.
 * Destroys all cells whose face matches any Wild Blocker face on the board.
 * Also destroys the Wild Blocker cells themselves.
 * Returns: { newGrid, destroyedFaces } for scoring.
 */
export function applyWildScatter(
  grid: Cell[][]
): { newGrid: Cell[][]; destroyedFaces: DieFace[] } {
  const wilds = getWildCells(grid);
  const scatterFaces = new Set(wilds.map(w => w.face));
  const destroyedFaces: DieFace[] = [];
  const newGrid = cloneGrid(grid);

  for (let r = 0; r < newGrid.length; r++) {
    for (let c = 0; c < newGrid[0].length; c++) {
      const cell = newGrid[r][c];
      if (
        cell.face !== null &&
        scatterFaces.has(cell.face) &&
        cell.state !== 'FROZEN' // Ice immune to scatter
      ) {
        if (cell.face !== null) destroyedFaces.push(cell.face);
        newGrid[r][c] = { ...cell, face: null, type: 'NONE', state: 'EMPTY' };
      }
    }
  }
  return { newGrid, destroyedFaces };
}
