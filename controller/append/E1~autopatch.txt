SECTION 3 — ENERGY TICK LOOP (server-side)
════════════════════════════════════════════════

Start energy tick interval (1000ms) on room creation:

  energyTick():
    For each connected player:
      if globalEnergyMode === 'PRIME':
        player.energy = Math.min(MAX_ENERGY, player.energy + PRIME_PASSIVE_GAIN)
      if globalEnergyMode === 'FRENZY':
        drain = player.energy < CATCHUP_THRESHOLD ? FRENZY_LOW_DRAIN : FRENZY_DRAIN
        player.energy = Math.max(0, player.energy - drain)
        if player.energy === 0:
          handlePlayerEnergyZero(player.id)

    updateGlobalMode():
      anyAbove = players.some(p => p.energy > FRENZY_THRESHOLD)
      allBelow = players.every(p => p.energy <= FRENZY_THRESHOLD)
      if anyAbove && globalEnergyMode !== 'FRENZY':
        globalEnergyMode = 'FRENZY'
        frenzyAnchorId = highest energy player id
        broadcast FRENZY_TRIGGERED { anchorId }
      if allBelow && globalEnergyMode !== 'PRIME':
        globalEnergyMode = 'PRIME'
        frenzyAnchorId = null
        broadcast PRIME_RETURNED {}

    Broadcast ENERGY_UPDATE to all players:
      { players: [{ id, energy, isAnchor, catchupActive }], globalMode }

submitChain(playerId: string, chain: GridPos[]):
  1. INPUT LOCK CHECK:
     if Date.now() - player.lastChainTimestamp < 100:
       return (anti-macro protection)
     player.lastChainTimestamp = Date.now()

  2. FRENZY CONFLICT RESOLUTION (server first-write-wins):
     During FRENZY: no activePlayerId check
     During PRIME: check playerId === activePlayerId

  3. CHAIN VALIDATION:
     Validate all positions are valid, non-empty cells
     Validate chain.length >= 2
     Extract faces (with 'WILD' for Wild Blocker cells)

  4. SCORING:
     result = scoreFarkle(faces, settings.threeOnesScore, settings.singleOneScore)
     resolvedFaces = result.resolvedFaces ?? faces

  5. ENERGY APPLICATION:
     const usedWild = faces.some(f => f === 'WILD')
     applyChainEnergy(player, chain.length, usedWild)
     updateGlobalMode()

  6. GRID UPDATE + CASCADE:
     Remove chained cells, damage adjacent blockers
     Run cascade: stepGravity → spawnTiles (SixPoolManager) → normalizeTiles
     After cascade: wildCount = scanForWilds(grid)

  7. WILD SCATTER CHECK:
     if wildCount >= WILD_BLOCKER_FRENZY_COUNT:
       { newGrid, destroyedFaces } = applyWildScatter(grid)
       scatterBonus = wildScatterScore(destroyedFaces)
       grid = newGrid
       Force all players' energy to FRENZY_THRESHOLD + 1
       broadcast WILD_SCATTER { destroyedFaces, scatterBonus }
       unbanked += scatterBonus
       wildCount = 0

  8. ROUND END CHECK (PRIME mode only):
     if result.isFarkle: handleRoundEnd(playerId, 'FARKLE')
     else if chain.length < 6 && globalEnergyMode === 'PRIME':
       handleRoundEnd(playerId, 'SHORT_CHAIN')
     else:
       // Continue round — 6-chain or FRENZY keeps going

  9. HEIST VAULT SPLIT (if HEIST mode):
     vaultContrib = Math.round(result.scaledScore * VAULT_SPLIT)
     personal = result.scaledScore - vaultContrib
     unbanked += personal
     heistState.vault += vaultContrib
     player.personalScore += personal
     player.vaultContribution += vaultContrib
     checkBetrayalThreshold(player)

  10. BROADCAST:
      CHAIN_RESULT { playerId, result, resolvedFaces }
      BOARD_UPDATE { grid }
      ENERGY_UPDATE { players energies, globalMode }
      if HEIST: VAULT_UPDATE { vault, personalScores, betrayalUnlocked }

initiateHeist(playerId):
  Validate: heistState.betrayalUnlocked[playerId] === true
  Validate: player.energy >= HEIST_ENERGY_COST
  player.energy -= HEIST_ENERGY_COST
  heistState.activeHeist = { initiatorId: playerId,
    countdownMs: HEIST_WINDOW_MS, blockerIds: [], startedAt: Date.now() }
  roundPhase = 'HEIST_WINDOW'
  broadcast HEIST_INITIATED { playerId, countdownMs: HEIST_WINDOW_MS }
  
  setTimeout(() => resolveHeist(), HEIST_WINDOW_MS)

blockHeist(playerId):
  if !heistState.activeHeist: return
  heistState.activeHeist.blockerIds.push(playerId)
  broadcast HEIST_BLOCKED { blockerId: playerId }

resolveHeist():
  const heist = heistState.activeHeist
  if !heist: return
  
  const archivist = getPlayerByRole('ARCHIVIST')
  const shieldedAmount = archivist
    ? Math.round(heistState.vault * ARCHIVIST_VAULT_SHIELD)
    : 0
  const stolenAmount = heistState.vault - shieldedAmount
  
  const initiator = players.get(heist.initiatorId)
  if initiator: initiator.banked += stolenAmount
  
  heistState.vault = shieldedAmount
  heistState.heistCount++
  heistState.successfulHeisters.push(heist.initiatorId)
  heistState.activeHeist = null
  roundPhase = 'ROLLING'
  
  broadcast HEIST_RESOLVED {
    winnerId: heist.initiatorId,
    stolenAmount,
    shieldedAmount,
    vaultReset: heistState.vault
  }

checkBetrayalThreshold(player):
  const unlocked = heistState.vault >= VAULT_THRESHOLD
                   && player.energy >= HEIST_ENERGY_COST
  heistState.betrayalUnlocked[player.id] = unlocked
  if unlocked: broadcast HEIST_AVAILABLE { playerId: player.id }
